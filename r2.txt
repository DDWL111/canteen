public  static Connection getConn(String dbDriver,String dbUrl,String us,String pw){

	      try {
			 Class.forName(dbDriver);
			 conn=DriverManager.getConnection(dbUrl,us,pw);
		} catch (ClassNotFoundException  e) {
			e.printStackTrace();
		}catch(SQLException e1){
			e1.printStackTrace();
		}
		return conn;
}
